<script>
    bubbles()
    confetti()
    hearts()
    fire()
    lines()

    function random(m, n) {
        return Math.floor(Math.random() * (n - m + 1)) + m;
    }

    function append(parent, text) {
        if (typeof text === 'string') {
            var temp = document.createElement('div');
            temp.innerHTML = text;
            // 防止元素太多 进行提速
            var frag = document.createDocumentFragment();
            while (temp.firstChild) {
                frag.appendChild(temp.firstChild);
            }
            parent.appendChild(frag);
        } else {
            parent.appendChild(text);
        }
    }


    function bubbles() {
        let bubbles = document.querySelector('.bubbles')
        let num = bubbles.offsetWidth / 5;
        let html = ''
        for (let i = 0; i <= num; i++) {
            let size = random(40, 80) / 10;
            let color = 'rgba(33, 150, 243, 0.5)'
            html += `
            <span class="particle" style="width:${size}px;
            height:${size}px;top:${ random(20,80)}%;left:${random(0,95)}%;background:${color};animation-delay:${random(0,30)/10}s"></span>
            `
        };
        append(bubbles, html)
    }

    function confetti() {
        let confetti = document.querySelector('.confetti')
        let num = confetti.offsetWidth / 5;
        let cols = ['#293b8b', '#ecf0f1', '#ececec', '#f5d76e', '#f4d03f'];
        let html = ''
        for (let i = 0; i <= num; i++) {
            let color = cols[parseInt(Math.random() * cols.length)];
            html += `
            <span class="particle" style="width:${random(6, 8)}px;
            height:${random(3, 4)}px;top:${ random(20,80)}%;left:${random(0,95)}%;background:${color};animation-delay:${random(0,30)/10}s"></span>
            `
        };
        append(confetti, html)
    }

    function hearts() {
        let hearts = document.querySelector('.hearts')
        let num = hearts.offsetWidth / 5;
        let size = (random(60, 120) / 10);
        let html = ''
        for (let i = 0; i <= num; i++) {
            html += `
            <span class="particle" style="width:${size}px;
            height:${size}px;
            top:${ random(20,80)}%;
            left:${random(0,95)}%;
            animation-delay:${random(0,30)/10}s"></span>
            `
        };
        append(hearts, html)
    }

    function fire() {
        let fire = document.querySelector('.fire')
        let num = fire.offsetWidth / 5;
        let size = random(8, 12);
        let html = ''
        for (let i = 0; i <= num; i++) {
            html += `
            <span class="particle" style="width:${size}px;
            height:${size}px;
            top:${ random(40,70)}%;
            left:${random(-10,100)}%;
            animation-delay:${random(0,30)/10}s"></span>
            `
        };
        append(fire, html)
    }

    function lines() {
        let lines = document.querySelector('.lines')
        let num = lines.offsetWidth / 5;
        let html = ''
        for (let i = 0; i <= num; i++) {
            html += `
            <span class="particle" style="width:${random(1,3)}px;
            height:${random(20,80)}px;
            top:${ random(-30,30)}%;
            left:${random(-10,110)}%;
            animation-delay:${random(0,30)/10}s"></span>
            `
        };
        append(lines, html)
    }
</script>
